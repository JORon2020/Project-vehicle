-- Table for Server Logs
CREATE TABLE server_logs (
    log_id INT AUTO_INCREMENT PRIMARY KEY,
    log_type VARCHAR(50),
    log_message TEXT,
    log_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Table for Traffic Metrics
CREATE TABLE traffic_metrics (
    metric_id INT AUTO_INCREMENT PRIMARY KEY,
    total_requests INT,
    unique_visitors INT,
    page_views INT,
    bandwidth_usage DECIMAL(10,2), -- Assuming bytes
    response_time DECIMAL(10,2), -- Assuming milliseconds
    peak_traffic_time TIMESTAMP
);

-- Table for Resource Utilization
CREATE TABLE resource_utilization (
    utilization_id INT AUTO_INCREMENT PRIMARY KEY,
    cpu_usage DECIMAL(5,2), -- Percentage
    memory_usage DECIMAL(5,2), -- Percentage
    disk_io DECIMAL(10,2), -- Assuming bytes
    network_bandwidth DECIMAL(10,2), -- Assuming bytes
    server_uptime INT -- In seconds
);

-- Table for HTTP Status Codes
CREATE TABLE http_status_codes (
    status_code INT PRIMARY KEY,
    status_description VARCHAR(100)
);

-- Table for Geolocation Data
CREATE TABLE geolocation_data (
    ip_address VARCHAR(45) PRIMARY KEY,
    country VARCHAR(100),
    region VARCHAR(100),
    city VARCHAR(100),
    latitude DECIMAL(10,6),
    longitude DECIMAL(10,6)
);

-- Table for Database Performance Metrics
CREATE TABLE database_performance_metrics (
    metric_id INT AUTO_INCREMENT PRIMARY KEY,
    database_queries INT,
    response_time DECIMAL(10,2), -- Assuming milliseconds
    connection_pool_usage INT,
    database_locks INT,
    query_execution_plans TEXT
);

-- Table for Server Configuration
CREATE TABLE server_configuration (
    configuration_id INT AUTO_INCREMENT PRIMARY KEY,
    server_software_version VARCHAR(50),
    web_server_configuration TEXT,
    database_configuration TEXT,
    application_framework_version VARCHAR(50),
    ssl_configuration TEXT
);

-- Table for Error Analysis
CREATE TABLE error_analysis (
    error_id INT AUTO_INCREMENT PRIMARY KEY,
    error_type VARCHAR(50),
    error_frequency INT,
    error_message TEXT,
    error_stack_trace TEXT
);
